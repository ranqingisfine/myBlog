9.2.1 з”ЁMicrometerд»ӘиЎЁеҢ–JVMеә”з”Ё

иЎҘе……дёӢcounterжҢҮж Үе’ҢgaugeжҢҮж Үзҡ„зҗҶи§Ј

зү№жҖ§CounterGauge
вҖҢж•°еҖјж–№еҗ‘вҖҢеҸӘеўһдёҚеҮҸпјҲжҲ–йҮҚзҪ®еҪ’йӣ¶пјүеҸҜеўһеҸҜеҮҸ
вҖҢе…ёеһӢз”ЁйҖ”вҖҢзҙҜз§ҜдәӢд»¶пјҲеҰӮиҜ·жұӮжҖ»ж•°пјүзһ¬ж—¶зҠ¶жҖҒпјҲеҰӮеҶ…еӯҳдҪҝз”ЁйҮҸпјү
вҖҢPromQL ж“ҚдҪңвҖҢrate(), increase()зӣҙжҺҘдҪҝз”ЁеҺҹе§ӢеҖј
вҖҢж•°жҚ®жЁЎеһӢвҖҢйҖӮеҗҲи®Ўз®—йҖҹзҺҮпјҲеҰӮ QPSпјүйҖӮеҗҲи§ӮеҜҹеҪ“еүҚеҖј

йӣҶжҲҗSpringBootзҡ„Prometheusдёүеү‘е®ўпјҡMicrometer+Prometheus+Grafana

MicrometerжҳҜPivotalжңҖжөҒиЎҢзҡ„зӣ‘жҺ§зі»з»ҹжҸҗдҫӣзҡ„дёҖдёӘз®ҖеҚ•зҡ„д»ӘиЎЁе®ўжҲ·з«Ҝй—ЁйқўжЁЎејҸпјҢе…Ғи®ёд»ӘиЎЁеҢ–JVMеә”з”ЁпјҢиҖҢж— йңҖе…іеҝғжҳҜйӮЈдёӘдҫӣеә”е•ҶжҸҗдҫӣзҡ„жҢҮж ҮгҖӮе®ғзҡ„дҪңз”Ёе’ҢSLF4Jзұ»дјјпјҢеҸӘдёҚиҝҮе®ғе…іжіЁзҡ„жҳҜеә”з”ЁжҢҮж ҮпјҢз®ҖиҖҢиЁҖд№Ӣе®ғжҳҜзӣ‘жҺ§з•Ңзҡ„SLF4jпјҢMicrometerд№ҹжҳҜдёҖдёӘMetricsзҡ„facadeпјҢе®ғжҳҜдёҖдёӘзӣ‘жҺ§жҢҮж Үзҡ„еәҰйҮҸзұ»еә“гҖӮ

ж—Ҙеёёз”ҹдә§дёӯпјҢжҺЁиҚҗдҪҝз”ЁMicrometerдҪңдёәжҢҮж Үй©ұеҠЁејҖеҸ‘зӣ‘жҺ§зҡ„й—ЁйқўгҖӮ

MicrometerдёӯйҮҚиҰҒзҡ„дёӨдёӘжҰӮеҝөжҳҜпјҡMeterе’ҢMeterRegistry

  • Meterпјҡз”ЁдәҺж”¶йӣҶеә”з”Ёзҡ„дёҖзі»еҲ—жҢҮж Үзҡ„жҺҘеҸЈгҖӮдҫӢеҰӮеўһйҮҸи®Ўж•°з”ЁCounterиЎЁзӨәпјҢеҚ•дёӘжҢҮж ҮеҖјз”ЁGaugeиЎЁзӨәпјҢи®Ўж—¶дәӢд»¶зҡ„ж¬Ўж•°е’ҢжҖ»ж—¶й—ҙз”ЁTimerиЎЁзӨәгҖӮжҜҸдёҖйЎ№жҢҮж ҮйғҪжңүдёҖдёӘе”ҜдёҖж ҮиҜҶзҡ„жҢҮж ҮеҗҚз§°е’Ңж Үзӯҫ
  • MeterRegistryпјҡMeterжҳҜз”ұMeterRegistryеҲӣе»әзҡ„гҖӮ

package com.example.component;

import io.micrometer.core.instrument.Counter;
import io.micrometer.core.instrument.Gauge;
import io.micrometer.core.instrument.MeterRegistry;
import io.micrometer.core.instrument.binder.MeterBinder;
import org.springframework.stereotype.Component;

import java.util.HashMap;
import java.util.Map;

/**
 * @Author Ran Qing
 * @Date 2025/5/17 15:08
 * жӯӨжЎҲдҫӢз»„д»¶еҶ…зҪ®дәҶдёҖдёӘи®Ўж•°еҷЁпјҢз”ЁдәҺеҜ№жңүи®Ўж•°еҷЁйңҖжұӮзҡ„дёҡеҠЎеңәжҷҜиҝӣиЎҢе·Ҙе…·е°ҒиЈ…
 */
@Component
public class DemoMetrcs implements MeterBinder {

    //CounterжҳҜдёҖдёӘеҸӘеўһдёҚеҮҸзҡ„и®Ўж•°еҷЁпјҢеҸҜд»Ҙз”ЁжқҘи®°еҪ•иҜ·жұӮжҲ–иҖ…й”ҷиҜҜзҡ„жҖ»йҮҸпјҢжҜ”еҰӮhttp_requests_totalзӯү
    public Counter counter;

    //GaugeеұһдәҺеҸҜеўһеҸҜеҮҸзҡ„д»ӘиЎЁзӣҳпјҢжӣҙеӨҡз”ЁжқҘеҸҚжҳ еә”з”ЁеҪ“еүҚзҡ„зҠ¶жҖҒпјҢжҜ”еҰӮдё»жңәеҪ“еүҚз©әй—Ізҡ„еҶ…еӯҳnode_memory_MemFreeжҲ–иҖ…е…¶д»–еұһжҖ§
    public Map<String,Double> map;

    DemoMetrcs(){
        map=new HashMap<>();
    }

    @Override
    public void bindTo(MeterRegistry meterRegistry) {
        //е®ҡд№үе№¶жіЁеҶҢдёҖдёӘеҗҚз§°дёәprometheus.demo.counterзҡ„и®Ўж•°еҷЁпјҢж ҮзӯҫжҳҜname:counter1
        this.counter=Counter.builder("prometheus.demo.counter")
                .tags(new String[]{"name","counter1"})
                .description("demo counter")
                .register(meterRegistry);

        //д»ҺдёҡеҠЎз«Ҝдј йҖ’зҡ„MapдёӯеҸ–еҮәдёҺkeyеҜ№еә”зҡ„еҖјж”ҫе…ҘжіЁеҶҢзҡ„Gaugeд»ӘиЎЁзӣҳдёӯпјҢж ҮзӯҫжҳҜname:gauge1
        Gauge.builder("prometheus.demo.gauge",map,x->x.get("x"))
                .tags("name","gauge1")
                .description("this is Gauge")
                .register(meterRegistry);

    }
}

package com.example.component;

import org.springframework.scheduling.annotation.Async;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;

/**
 * @Author Ran Qing
 * @Date 2025/5/17 15:17
 */
@Component
public class SimulationRequest {

    private Integer count1=0;

    @Resource
    private DemoMetrcs demoMetrcs;

    @Async("One")
    @Scheduled(fixedDelay = 1000)
    public void increment1(){
        count1++;
        demoMetrcs.counter.increment();
        demoMetrcs.map.put("x",Double.valueOf(count1));
        System.out.println("increment1 count:"+count1);
    }
}
