class Solution {
    public String convert(String s, int numRows) {
        if(numRows < 2){
            return s;
        }
        //еҲқе§ӢеҢ–иЎҢеӯ—з¬ҰдёІ
        List<StringBuilder> raws=new ArrayList<StringBuilder>();
        for(int i=0;i<numRows;i++){
            raws.add(new StringBuilder());
        }
        //еҲқе§ӢеҢ–иЎҢдёӢж Үе’Ңж ҮиҜҶ
        int i=0;
        int flag=-1;
        for(char c:s.toCharArray()){
            raws.get(i).append(c);
            //еҰӮжһңжҳҜз¬¬1иЎҢжҲ–иҖ…жңҖеҗҺ1иЎҢпјҢеҲҷж”№еҸҳж ҮиҜҶж–№дҪҚ
            if(i==0 || numRows-1==i){
                flag=-flag;
            }
            i+=flag;
        }
        StringBuilder sb=new StringBuilder();
        for(StringBuilder x:raws){
            sb.append(x.toString());
        }
        return sb.toString();
    }
}
